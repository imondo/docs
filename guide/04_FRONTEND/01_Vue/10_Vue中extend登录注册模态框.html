    instance.doClose();
  }
  instance = new LoginModal({
    data: {
      show: true,
      ...options
    }
  });
  instance.ok = () => {
    return new Promise(resolve => {
      const before = options.ok ? options.ok() : false;
      if (before && before.then) {
        before.then(
          () => resolve(true),
          () => {
            console.log("reject");
          }
        );
      } else if (typeof before === "boolean" && before !== false) {
        resolve(true);
      }
    });

  document.body.appendChild(instance.$el);
  return instance;
};

  return ModalBox({
    type: "sigin",
    ok

};

  return ModalBox({
    type: "register",
    ok
  });

  instance.doClose();
  instance.show = false;
