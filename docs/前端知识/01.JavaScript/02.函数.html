# 函数中的this

  1. this 是什么

    • 任何函数本质上都是通过某个对象调用的

    • 所有函数内部都有一个变量 this

    • 它的值是调用函数的当前对象

  2. 如何确顶 this 的值

    • test(): window
    • p.test(): p
    • new test(): 新创建的对象
    • p.call(obj): obj
function Person() {
	console.log(this)
	this.color = function() {
		console.log(this)
	}
    this.setColor = function() {
		console.log(this)
	}
}
Person() // this 指向 window
var p = new Person() // this 指向 p
p.color() // this 指向 p

var obj = {}
p.setColor.call(obj) // this 指向 obj

var test = p.setColor;
test() // this 指向 window

function fun1() {
    function fun2() {
        console.log(this)
    }
    
    fun2() // this 指向 window
}
